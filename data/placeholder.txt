kpis.csv
SELECT
  COUNT(*) AS num_registros,
  SUM(unidades_vendida) AS total_unidades,
  ROUND(SUM(beneficio_total_eur), 2) AS beneficio_total,
  ROUND(AVG(beneficio_total_eur), 2) AS beneficio_medio
FROM tienda_limpia;

beneficio_por_nivel.csv
SELECT
  Nivel_de_rentabi AS nivel,
  ROUND(SUM(beneficio_total_eur), 2) AS beneficio_total
FROM tienda_limpia
GROUP BY Nivel_de_rentabi
ORDER BY beneficio_total DESC;

ventas_por_estado.csv
SELECT
  Estado,
  COUNT(*) AS num_ventas,
  SUM(Unidades_vendida) AS unidades
FROM tienda_limpia
GROUP BY Estado
ORDER BY num_ventas DESC;

top_productos_beneficio.csv
SELECT
  Producto,
  ROUND(SUM(beneficio_total_eur), 2) AS beneficio_total,
  SUM(Unidades_vendida) AS unidades
FROM tienda_limpia
GROUP BY Producto
ORDER BY beneficio_total DESC
LIMIT 10;

detalle.csv
SELECT
  estado AS estado,
  nivel_de_rentabi AS nivel,
  producto AS producto,
  unidades_vendida AS unidades,
  Beneficio_total_eur AS beneficio_total
FROM tienda_limpia;
